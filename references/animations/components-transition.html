Animations

Components - Transition

Transition applies a Svelte Transition to a container on a container element, that's linked to the playback system.

# Imports

<script>
    import {Transition} from "@svelte-in-motion/animations";
</script>

# Definitions

<Transition
    element={HTMLDivElement | HTMLSpanElement | null}

    class={string}
    is={"div" | "span"}

    delay={number}
    duration={number}
    easing={(completion: number) => number}

    transition={ITransition}

    inverse={boolean}
    parameters={object}
/>

# Usage

<script>
    import {Transition, fade} from "@svelte-in-motion/animations";
</script>

<!-- Entrance Animation -->

<Transition
    transition={fade}
>
    ...
</Transition>

<!-- Exit Animation -->

<Transition
    transition={fade}
    inverse
>
    ...
</Transition>
